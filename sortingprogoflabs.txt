{
         int k;
        for(i=0;i<n-1;i++)
        {
        printf("\n\n PASS->%d ",i+1);
        for(j=i+1;j<=n-1;j++)
        {
                if(a[i]>a[j])
                {
                        temp=a[i];
                        a[i]=a[j];
                        a[j]=temp;
                }
        }
        for(k=0;k<n;k++)
        printf("%d ",a[k]);
        printf("\n");
        }
}
